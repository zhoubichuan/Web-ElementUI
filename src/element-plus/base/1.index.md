```vue
<template>
  <!-- 搜索组件 -->
  <web-search
    class="search-part"
    :condition="condition"
    :value="searchField"
    @input="(val) => updateField(val)"
    @handleSearch="handleSearch"
  >
  </web-search>
</template>
<script>
export default {
  name: "DataEntitySearch",
  data() {
    return {
      searchField: {},
      condition: [],
    };
  },
  watch: {},
  created() {
    this.querySearchTemplate();
  },
  methods: {
    async querySearchTemplate(condition = {}) {
      let { data } = await axios.get(this.$withBase("/data/search.json"));
      if (data) {
        let { result } = data;
        this.condition = result;
      } else {
        this.condition = [];
      }
    },
    updateField(val) {
      this.searchField = val;
      this.$emit("input", this.searchField);
    },
    handleSearch(condition) {
      this.$emit("handleSearch", condition);
    },
  },
};
</script>
```
